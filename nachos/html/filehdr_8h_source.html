49// -------------------------------------------------
50// Representation of file/directory headers on disk
51// -------------------------------------------------
52//
53// 1. First header sector
54//
55// .----------------------.
56// | isDir | if it is a directory, 1, 0 otherwise
57// | numBytes | total size of the data (header excluded)
58// | numSectors | total number of sectors
59// | numHeaderSectors | number of header sectors
60// .----------------------.
61// | List of the | The list of the sector addresses containing data
62// | Data sectors | (at most DatasInFirstSector sectors)
63// | |
64// . ---------------------.
65// | Next header sector | The sector containing the remaining of the
66// | | list of data sectors (a "normal" header sector,
67// .----------------------. see below)
68//
69// 2. The other "normal" header sectors
70//
71// .----------------------.
72// | List of the | The list of the sector addresses containing data
73// | Data sectors (ctd.) | (at most DatasInSector sectors)
74// | |
75// .----------------------.
76// | Next header sector | The sector containing the remaining of the
77// | | list of data sectors (a "normal" header sector,
78// .----------------------. see below)
79//
80// Be careful when modifying the format of the file header on disk
81// Methods FetchFrom and WriteBack assume THIS representation
82
83// Number of sector numbers that can be stored in the first sector
84// representing a file header, which contains a header of 4 ints
85// and a trailer of 1 int (5 integers in total)
86#define DatasInFirstSector \
87 ((int)((g_cfg->SectorSize - 5*sizeof(int)) / sizeof(int)))
88
89// Number of sector numbers that can be put in a "normal" header sector
90#define DatasInSector \
91 ((int)((g_cfg->SectorSize - 1*sizeof(int)) / sizeof(int)))
92
93// Get the value of the next header sector, given the hdrSector array
94// of int representing the contents of the current header sector
95#define NextHeaderSector(hdrSector) \
96 (((int*)(hdrSector))[(g_cfg->SectorSize / sizeof(int)) - 1])
97
99#define MAX_HEADER_SECTORS 32

102// (computed according to the disk representation of the file)
103#define MAX_DATA_SECTORS \
104 ((int)((MAX_HEADER_SECTORS-1)*DatasInSector + DatasInFirstSector))
105
106#define MAX_FILE_LENGTH \
107 ((int)((MAX_DATA_SECTORS) * g_cfg->SectorSize))
